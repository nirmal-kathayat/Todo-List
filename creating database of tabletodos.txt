//creating database of table:todos   for website
//database name : login_db

CREATE TABLE todos(
	id int AUTO_INCREMENT PRIMARY KEY,
	title VARCHAR(250) NULL,
	status int(3) NULL,
	datetime(5)
);

//creating table for signup system: user

CREATE TABLE user(
	id int AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(250) ,
	email VARCHAR(255) ,
	password_hash VARCHAR(255)	
);
